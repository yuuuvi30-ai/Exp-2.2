<body>
    <div class="card" id="card">
        <h1 class="title">Live Character Counter</h1>
        <p class="subtitle">Type in the box below. The counter updates instantly and shows warnings near the limit.</p>

        <label for="text">Your Text</label>
        <textarea id="text" placeholder="Start typing..."></textarea>

        <div class="row">
            <div class="counter" id="counter">
                <strong id="count">0</strong>
                <span>/</span>
                <span id="limitText">200</span>
            </div>
            <div class="hint" id="hint">Keep it under the limit for best readability.</div>
        </div>

        <div class="barWrap" aria-hidden="true">
            <div class="bar" id="bar"></div>
        </div>
    </div>

    <script>
        const LIMIT = 200;

        const card = document.getElementById("card");
        const text = document.getElementById("text");
        const countEl = document.getElementById("count");
        const limitText = document.getElementById("limitText");
        const hint = document.getElementById("hint");
        const bar = document.getElementById("bar");

        limitText.textContent = String(LIMIT);

        function update() {
            const len = text.value.length;
            countEl.textContent = String(len);

            const pct = Math.min(100, (len / LIMIT) * 100);
            bar.style.width = pct + "%";

            card.classList.remove("warn", "danger");

            if (len >= LIMIT) {
                card.classList.add("danger");
                hint.textContent = "Limit reached. Remove a few characters.";
            } else if (len >= Math.floor(LIMIT * 0.8)) {
                card.classList.add("warn");
                hint.textContent = "You are close to the limit.";
            } else {
                hint.textContent = "Keep it under the limit for best readability.";
            }
        }

        text.addEventListener("input", update);
        update();
    </script>
